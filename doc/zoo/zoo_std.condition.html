  Lemma condition_wait_until𑁒spec' Ψ t mtx pred P :
    {{{
      condition_inv t ∗
      mutex_inv mtx P ∗
      mutex_locked mtx ∗
      P ∗

      □ (
        mutex_locked mtx -∗
        P -∗
        Ψ false -∗
