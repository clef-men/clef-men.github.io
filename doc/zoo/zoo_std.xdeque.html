Section zoo_G.
  Context `{zoo_G : !ZooG Σ}.

  Definition xdeque_model t nodes : iProp Σ :=
    ∃ l,
    ⌜t = #l⌝ ∗
    l.[xdeque_prev] ↦ from_option #@{location} t (last nodes) ∗
    l.[xdeque_next] ↦ from_option #@{location} t (head nodes) ∗
    xdlchain t nodes t.

  #[global] Instance xdeque_model_timeless t nodes :
    Timeless (xdeque_model t nodes).

  Lemma xdeque_model_exclusive t nodes1 nodes2 :
    xdeque_model t nodes1 -∗
    xdeque_model t nodes2 -∗
    False.

  Lemma xdeque_model_NoDup t nodes :
    xdeque_model t nodes ⊢
    ⌜NoDup nodes⌝.

  Lemma xdeque_create𑁒spec :
    {{{
      True
    }}}
      xdeque_create ()
    {{{
      t
    , RET t;
      (∃ l, ⌜t = #l⌝ ∗ meta_token l ⊤) ∗
      xdeque_model t []
    }}}.

  Lemma xdeque_is_empty𑁒spec t nodes :
    {{{
      xdeque_model t nodes
    }}}
      xdeque_is_empty t
    {{{
      RET #(bool_decide (nodes = []%list));
      xdeque_model t nodes
    }}}.

  #[local] Lemma xdeque_link𑁒spec node1 v1 node2 v2 :
    {{{
      node1.[xdeque_next] ↦ v1 ∗
      node2.[xdeque_prev] ↦ v2
    }}}
      xdeque_link #node1 #node2
    {{{
      RET ();
      node1.[xdeque_next] ↦ #node2 ∗
      node2.[xdeque_prev] ↦ #node1
    }}}.

  Lemma xdeque_push_front𑁒spec t nodes node prev next :
    {{{
      xdeque_model t nodes ∗
      node.[xdeque_prev] ↦ prev ∗
      node.[xdeque_next] ↦ next
    }}}
      xdeque_push_front t #node
    {{{
      RET ();
      xdeque_model t (node :: nodes)
    }}}.

  Lemma xdeque_push_back𑁒spec t nodes node prev next :
    {{{
      xdeque_model t nodes ∗
      node.[xdeque_prev] ↦ prev ∗
      node.[xdeque_next] ↦ next
    }}}
      xdeque_push_back t #node
    {{{
      RET ();
      xdeque_model t (nodes ++ [node])
    }}}.

  Lemma xdeque_pop_front𑁒spec t nodes :
    {{{
      xdeque_model t nodes
    }}}
      xdeque_pop_front t
    {{{
      RET #*@{location} $ head nodes : option val;
      xdeque_model t (tail nodes)
    }}}.

  Lemma xdeque_pop_back𑁒spec t nodes :
    {{{
      xdeque_model t nodes
    }}}
      xdeque_pop_back t
    {{{
      o
    , RET #*@{location} o : option val;
      match o with
      | None ⇒
          ⌜nodes = []⌝ ∗
          xdeque_model t []
      | Some node ⇒
          ∃ nodes',
          ⌜nodes = nodes' ++ [node]⌝ ∗
          xdeque_model t nodes'
      end
    }}}.

  Lemma xdeque_remove𑁒spec {t nodes} i node :
    nodes !! i = Some node →
    {{{
      xdeque_model t nodes
    }}}
      xdeque_remove #node
    {{{
      RET ();
      xdeque_model t (delete i nodes)
    }}}.

  #[local] Lemma xdeque_iter_aux𑁒spec Ψ i fn l nodes node :
    (nodes ++ [l]) !! i = Some node →
    {{{
      ▷ Ψ (take i nodes) ∗
      xdeque_model #l nodes ∗
      □ (
        ∀ nodes_done node nodes_todo,
        ⌜nodes = nodes_done ++ node :: nodes_todo⌝ -∗
        Ψ nodes_done -∗
        WP fn #node {{ res,
          ⌜res = ()%V⌝ ∗
          ▷ Ψ (nodes_done ++ [#node])
        }}
      )
    }}}
      xdeque_iter_aux fn #l #node
    {{{
      RET ();
      xdeque_model #l nodes ∗
      Ψ nodes
    }}}.
  Lemma xdeque_iter𑁒spec Ψ fn t nodes :
    {{{
      ▷ Ψ [] ∗
      xdeque_model t nodes ∗
      □ (
        ∀ nodes_done node nodes_todo,
        ⌜nodes = nodes_done ++ node :: nodes_todo⌝ -∗
        Ψ nodes_done -∗
        WP fn #node {{ res,
          ⌜res = ()%V⌝ ∗
          ▷ Ψ (nodes_done ++ [#node])
        }}
      )
    }}}
      xdeque_iter fn t
    {{{
      RET ();
      xdeque_model t nodes ∗
      Ψ nodes
    }}}.
End zoo_G.
