  #[local] Definition model' t vs extra : iProp Σ :=
    ∃ l data,
    ⌜t = #l⌝ ∗
    l.[size] ↦ #(length vs) ∗

    array_model data (DfracOwn 1) (vs ++ replicate extra ()%V).
