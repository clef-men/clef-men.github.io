  #[local] Definition model' t vs extra : iProp Σ :=
    ∃ l data,
    ⌜t = #l⌝ ∗
    l.[size] ↦ #(length vs) ∗
    l.[data] ↦ data ∗
    array_model data (DfracOwn 1) (vs ++ replicate extra ()%V).
  #[local] Instance : CustomIpat "model'" :=
    " ( %l{} & %data{} & -> & Hl{}_size & Hl{}_data & Hmodel ) ".
  Definition dynarray_1_model t vs : iProp Σ :=
    ∃ extra,
