Library examples.fibonacci

Require Import zoo.prelude.
Require Import zoo.options.

Fixpoint fibonacci n :=
  match n with
  | 0 ⇒
      0
  | ˖n ⇒
      match n with
      | 0 ⇒
          1
      | ˖m ⇒
          fibonacci n + fibonacci m
      end
  end.
#[global] Arguments fibonacci !_ /.

Lemma fibonacci𑁒spec n :
  fibonacci n =
    if decide (n ≤ 1) then
      n
    else
      fibonacci (n - 1) + fibonacci (n - 2).
Lemma fibonacci𑁒spec𑁒Z n :
  (0 ≤ n)%Z →
  fibonacci ₊n =
    if decide (n ≤ 1)%Z then
      ₊n
    else
      fibonacci ₊(n - 1) + fibonacci ₊(n - 2).

Lemma fibonacci𑁒base n :
  n ≤ 1 →
  fibonacci n = n.
Lemma fibonacci𑁒recursive n :
  1 < n →
  fibonacci n = fibonacci (n - 1) + fibonacci (n - 2).
