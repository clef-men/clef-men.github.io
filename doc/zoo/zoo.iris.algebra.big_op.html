Section big_unionL.
  Context {SI : sidx}.
  Context {A : Type}.
  Context `{Countable K}.

  Implicit Type x : A.
  Implicit Type l : list A.
  Implicit Type f : nat → A → gset K.

  Lemma big_unionL𑁒elem_of {f} y l :
    y ∈ ([∪ list] k ↦ x ∈ l, f k x) →
      ∃ i x,
      l !! i = Some x ∧
      y ∈ f i x.
End big_unionL.

Section big_unionM.
  Context {SI : sidx}.
  Context `{Countable K}.
  Context {A : Type}.
  Context `{Countable B}.

  Implicit Type k : K.
  Implicit Type x : A.
  Implicit Type y : B.
  Implicit Type m : gmap K A.
  Implicit Type f : K → A → gset B.

  Lemma big_unionM𑁒elem_of {f} y m :
    y ∈ ([∪ map] k ↦ x ∈ m, f k x) →
      ∃ k x,
      m !! k = Some x ∧
      y ∈ f k x.
End big_unionM.
