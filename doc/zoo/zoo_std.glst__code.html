Definition glst_rev_app : val :=
  rec: "rev_app" "t1" "t2" ⇒
    match: "t1" with
    | Gnil ⇒
        "t2"
    | Gcons "v" "t1" ⇒

    end.

Definition glst_rev : val :=
  fun: "t" ⇒
    glst_rev_app "t" §Gnil.
