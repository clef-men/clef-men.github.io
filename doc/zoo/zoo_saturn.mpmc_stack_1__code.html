  rec: "pop" "t" ⇒
    match: !"t" with
    | Gnil ⇒
        §None
    | Gcons "v" "new_" as "old" ⇒
        if: CAS "t".[contents] "old" "new_" then (
          ‘Some( "v" )
        ) else (
          domain_yield () ;;
          "pop" "t"
        )
    end.
