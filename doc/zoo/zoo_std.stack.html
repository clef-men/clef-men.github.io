Section zoo_G.
  Context `{zoo_G : !ZooG Σ}.

  Definition stack_model t vs :=
    dynarray_1_model t (reverse vs).

  #[global] Instance stack_model_timeless t vs :
    Timeless (stack_model t vs).

  Lemma stack_make𑁒spec :
    {{{
      True
    }}}
      stack_create ()
    {{{
      t
    , RET t;
      stack_model t []
    }}}.

  Lemma stack_is_empty𑁒spec t vs :
    {{{
      stack_model t vs
    }}}
      stack_is_empty t
    {{{
      RET #(bool_decide (vs = []%list));
      stack_model t vs
    }}}.

  Lemma stack_push𑁒spec t vs v :
    {{{
      stack_model t vs
    }}}
      stack_push t v
    {{{
      RET ();
      stack_model t (v :: vs)
    }}}.

  Lemma stack_pop𑁒spec {t vs} v vs' :
    vs = v :: vs' →
    {{{
      stack_model t vs
    }}}
      stack_pop t
    {{{
      RET v;
      stack_model t vs'
    }}}.
End zoo_G.
