  Section chunk۰model.
    Definition chunk۰model l dq vs : iProp Σ :=
      l ↦∗{dq} vs.

    #[global] Instance chunk۰model𑁒timeless l dq vs :
      Timeless (chunk۰model l dq vs).

    #[global] Instance chunk۰model𑁒persistent l vs :
      Persistent (chunk۰model l DfracDiscarded vs).

    #[global] Instance chunk۰model𑁒fractional l vs :
      Fractional (λ q, chunk۰model l (DfracOwn q) vs).
    #[global] Instance chunk۰model𑁒as_fractional l q vs :
      AsFractional (chunk۰model l (DfracOwn q) vs) (λ q, chunk۰model l (DfracOwn q) vs) q.

    Lemma chunk۰model𑁒nil l dq :
      ⊢ chunk۰model l dq [].

    Lemma chunk۰model𑁒singleton l dq v :
      l ↦{dq} v ⊣⊢
      chunk۰model l dq [v].
    Lemma chunk۰model𑁒singleton₁ l dq v :
      l ↦{dq} v ⊢
      chunk۰model l dq [v].
    Lemma chunk۰model𑁒singleton₂ l dq v :
      chunk۰model l dq [v] ⊢
      l ↦{dq} v.

    Lemma chunk۰model𑁒app l dq vs1 vs2 :
      chunk۰model l dq vs1 ∗
      chunk۰model (l +ₗ length vs1) dq vs2 ⊣⊢
      chunk۰model l dq (vs1 ++ vs2).
    Lemma chunk۰model𑁒app₁ dq l1 vs1 l2 vs2 :
      l2 = l1 +ₗ length vs1 →
      chunk۰model l1 dq vs1 -∗
      chunk۰model l2 dq vs2 -∗
      chunk۰model l1 dq (vs1 ++ vs2).
    Lemma chunk۰model𑁒app₂ {l dq vs} vs1 vs2 :
      vs = vs1 ++ vs2 →
      chunk۰model l dq vs ⊢
        chunk۰model l dq vs1 ∗
        chunk۰model (l +ₗ length vs1) dq vs2.

    Lemma chunk۰model𑁒app𑁒3 l dq vs1 vs2 vs3 :
      chunk۰model l dq vs1 ∗
      chunk۰model (l +ₗ length vs1) dq vs2 ∗
      chunk۰model (l +ₗ ⁺(length vs1 + length vs2)) dq vs3 ⊣⊢
      chunk۰model l dq (vs1 ++ vs2 ++ vs3).
    Lemma chunk۰model𑁒app𑁒3₁ dq l1 vs1 l2 vs2 l3 vs3 :
      l2 = l1 +ₗ length vs1 →
      l3 = l1 +ₗ ⁺(length vs1 + length vs2) →
      chunk۰model l1 dq vs1 -∗
      chunk۰model l2 dq vs2 -∗
      chunk۰model l3 dq vs3 -∗
      chunk۰model l1 dq (vs1 ++ vs2 ++ vs3).
    Lemma chunk۰model𑁒app𑁒3₂ {l dq vs} vs1 vs2 vs3 :
      vs = vs1 ++ vs2 ++ vs3 →
      chunk۰model l dq vs ⊢
        chunk۰model l dq vs1 ∗
        chunk۰model (l +ₗ length vs1) dq vs2 ∗
        chunk۰model (l +ₗ ⁺(length vs1 + length vs2)) dq vs3.

    Lemma chunk۰model𑁒cons l dq v vs :
      l ↦{dq} v ∗
      chunk۰model (l +ₗ 1) dq vs ⊣⊢
      chunk۰model l dq (v :: vs).
    Lemma chunk۰model𑁒cons₁ l dq v vs :
      l ↦{dq} v -∗
      chunk۰model (l +ₗ 1) dq vs -∗
      chunk۰model l dq (v :: vs).
    Lemma chunk۰model𑁒cons₂ l dq v vs :
      chunk۰model l dq (v :: vs) ⊢
        l ↦{dq} v ∗
        chunk۰model (l +ₗ 1) dq vs.
    #[global] Instance chunk۰model𑁒cons𑁒frame l dq v vs R Q :
      Frame false R (l ↦{dq} v ∗ chunk۰model (l +ₗ 1) dq vs) Q →
      Frame false R (chunk۰model l dq (v :: vs)) Q
    | 2.

    Lemma chunk۰model𑁒update {l dq vs} (i : Z) i_ v :
      (0 ≤ i)%Z →
      vs !! i_ = Some v →
      i_ = ₊i →
      chunk۰model l dq vs ⊢
        (l +ₗ i) ↦{dq} v ∗
        ( ∀ w,
          (l +ₗ i) ↦{dq} w -∗
          chunk۰model l dq (<[i_ := w]> vs)
        ).
    Lemma chunk۰model𑁒lookup𑁒acc {l dq vs} (i : Z) i_ v :
      (0 ≤ i)%Z →
      vs !! i_ = Some v →
      i_ = ₊i →
      chunk۰model l dq vs ⊢
        (l +ₗ i) ↦{dq} v ∗
        ( (l +ₗ i) ↦{dq} v -∗
          chunk۰model l dq vs
        ).
    Lemma chunk۰model𑁒lookup {l dq vs} (i : Z) i_ v :
      (0 ≤ i)%Z →
      vs !! i_ = Some v →
      i_ = ₊i →
      chunk۰model l dq vs ⊢
      (l +ₗ i) ↦{dq} v.

    Lemma chunk۰model𑁒update' {l} {i : Z} {dq vs} j k v :
      (0 ≤ i ≤ j)%Z →
      vs !! k = Some v →
      k = ₊j - ₊i →
      chunk۰model (l +ₗ i) dq vs ⊢
        (l +ₗ j) ↦{dq} v ∗
        ( ∀ w,
          (l +ₗ j) ↦{dq} w -∗
          chunk۰model (l +ₗ i) dq (<[k := w]> vs)
        ).
    Lemma chunk۰model𑁒lookup𑁒acc' {l} {i : Z} {dq vs} j k v :
      (0 ≤ i ≤ j)%Z →
      vs !! k = Some v →
      k = ₊j - ₊i →
      chunk۰model (l +ₗ i) dq vs ⊢
        (l +ₗ j) ↦{dq} v ∗
        ( (l +ₗ j) ↦{dq} v -∗
          chunk۰model (l +ₗ i) dq vs
        ).
    Lemma chunk۰model𑁒lookup' {l} {i : Z} {dq vs} j k v :
      (0 ≤ i ≤ j)%Z →
      vs !! k = Some v →
      k = ₊j - ₊i →
      chunk۰model (l +ₗ i) dq vs ⊢
      (l +ₗ j) ↦{dq} v.

    Lemma chunk۰model𑁒valid l dq vs :
      0 < length vs →
      chunk۰model l dq vs ⊢
      ⌜✓ dq⌝.
    Lemma chunk۰model𑁒combine l dq1 vs1 dq2 vs2 :
      length vs1 = length vs2 →
      chunk۰model l dq1 vs1 -∗
      chunk۰model l dq2 vs2 -∗
        ⌜vs1 = vs2⌝ ∗
        chunk۰model l (dq1 ⋅ dq2) vs1.
    Lemma chunk۰model𑁒valid𑁒2 l dq1 vs1 dq2 vs2 :
      0 < length vs1 →
      length vs1 = length vs2 →
      chunk۰model l dq1 vs1 -∗
      chunk۰model l dq2 vs2 -∗
        ⌜✓ (dq1 ⋅ dq2)⌝ ∗
        ⌜vs1 = vs2⌝.
    Lemma chunk۰model𑁒agree l dq1 vs1 dq2 vs2 :
      length vs1 = length vs2 →
      chunk۰model l dq1 vs1 -∗
      chunk۰model l dq2 vs2 -∗
      ⌜vs1 = vs2⌝.
    Lemma chunk۰model𑁒dfrac𑁒ne l1 dq1 vs1 l2 dq2 vs2 :
      0 < length vs1 →
      length vs1 = length vs2 →
      ¬ ✓ (dq1 ⋅ dq2) →
      chunk۰model l1 dq1 vs1 -∗
      chunk۰model l2 dq2 vs2 -∗
      ⌜l1 ≠ l2⌝.
    Lemma chunk۰model𑁒ne l1 vs1 l2 dq2 vs2 :
      0 < length vs1 →
      length vs1 = length vs2 →
      chunk۰model l1 (DfracOwn 1) vs1 -∗
      chunk۰model l2 dq2 vs2 -∗
      ⌜l1 ≠ l2⌝.
    Lemma chunk۰model𑁒exclusive l vs1 dq2 vs2 :
      0 < length vs1 →
      length vs1 = length vs2 →
      chunk۰model l (DfracOwn 1) vs1 -∗
      chunk۰model l dq2 vs2 -∗
      False.
    Lemma chunk۰model𑁒persist l dq vs :
      chunk۰model l dq vs ⊢ |==>
      chunk۰model l DfracDiscarded vs.
  End chunk۰model.

  Section chunk۰span.
    Definition chunk۰span l dq n : iProp Σ :=
      ∃ vs,
      ⌜length vs = n⌝ ∗
      chunk۰model l dq vs.

    #[global] Instance chunk۰span𑁒timeless l dq n :
      Timeless (chunk۰span l dq n).

    #[global] Instance chunk۰span𑁒persistent l n :
      Persistent (chunk۰span l DfracDiscarded n).

    #[global] Instance chunk۰span𑁒fractional l n :
      Fractional (λ q, chunk۰span l (DfracOwn q) n).
    #[global] Instance chunk۰span𑁒as_fractional l q n :
      AsFractional (chunk۰span l (DfracOwn q) n) (λ q, chunk۰span l (DfracOwn q) n) q.

    Lemma chunk۰span𑁒singleton l dq :
      ( ∃ v,
        l ↦{dq} v
      ) ⊣⊢
      chunk۰span l dq 1.
    Lemma chunk۰span𑁒singleton₁ l dq v :
      l ↦{dq} v ⊢
      chunk۰span l dq 1.
    Lemma chunk۰span𑁒singleton₂ l dq :
      chunk۰span l dq 1 ⊢
        ∃ v,
        l ↦{dq} v.

    Lemma chunk۰span𑁒cons l dq n :
      ( ∃ v,
        l ↦{dq} v ∗
        chunk۰span (l +ₗ 1) dq n
      ) ⊣⊢
      chunk۰span l dq ˖n.
    Lemma chunk۰span𑁒cons₁ l dq v n :
      l ↦{dq} v -∗
      chunk۰span (l +ₗ 1) dq n -∗
      chunk۰span l dq ˖n.
    Lemma chunk۰span𑁒cons₂ l dq n :
      chunk۰span l dq ˖n ⊢
        ∃ v,
        l ↦{dq} v ∗
        chunk۰span (l +ₗ 1) dq n.
    #[global] Instance chunk۰span𑁒cons𑁒frame l dq v n R Q :
      Frame false R (l ↦{dq} v ∗ chunk۰span (l +ₗ 1) dq n) Q →
      Frame false R (chunk۰span l dq ˖n) Q
    | 2.

    Lemma chunk۰span𑁒app l dq n1 n2 :
      chunk۰span l dq n1 ∗
      chunk۰span (l +ₗ n1) dq n2 ⊣⊢
      chunk۰span l dq (n1 + n2).
    Lemma chunk۰span𑁒app₁ dq l1 (n1 : nat) l2 n2 :
      l2 = l1 +ₗ n1 →
      chunk۰span l1 dq n1 -∗
      chunk۰span l2 dq n2 -∗
      chunk۰span l1 dq (n1 + n2).
    Lemma chunk۰span𑁒app₂ {l dq n} n1 n2 :
      n = n1 + n2 →
      chunk۰span l dq n ⊢
        chunk۰span l dq n1 ∗
        chunk۰span (l +ₗ n1) dq n2.

    Lemma chunk۰span𑁒app𑁒3 l dq n1 (n2 : nat) n3 :
      chunk۰span l dq n1 ∗
      chunk۰span (l +ₗ n1) dq n2 ∗
      chunk۰span (l +ₗ ⁺(n1 + n2)) dq n3 ⊣⊢
      chunk۰span l dq (n1 + n2 + n3).
    Lemma chunk۰span𑁒app𑁒3₁ dq l1 n1 l2 n2 l3 n3 :
      l2 = l1 +ₗ n1 →
      l3 = l1 +ₗ ⁺(n1 + n2) →
      chunk۰span l1 dq n1 -∗
      chunk۰span l2 dq n2 -∗
      chunk۰span l3 dq n3 -∗
      chunk۰span l1 dq (n1 + n2 + n3).
    Lemma chunk۰span𑁒app𑁒3₂ {l dq n} n1 n2 n3 :
      n = n1 + n2 + n3 →
      chunk۰span l dq n ⊢
        chunk۰span l dq n1 ∗
        chunk۰span (l +ₗ n1) dq n2 ∗
        chunk۰span (l +ₗ ⁺(n1 + n2)) dq n3.

    Lemma chunk۰span𑁒update {l dq n} (i : Z) :
      (0 ≤ i < n)%Z →
      chunk۰span l dq n ⊢
        ∃ v,
        (l +ₗ i) ↦{dq} v ∗
        ( ∀ w,
          (l +ₗ i) ↦{dq} w -∗
          chunk۰span l dq n
        ).
    Lemma chunk۰span𑁒lookup𑁒acc {l dq n} (i : Z) :
      (0 ≤ i < n)%Z →
      chunk۰span l dq n ⊢
        ∃ v,
        (l +ₗ i) ↦{dq} v ∗
        ( (l +ₗ i) ↦{dq} v -∗
          chunk۰span l dq n
        ).
    Lemma chunk۰span𑁒lookup {l dq n} (i : Z) :
      (0 ≤ i < n)%Z →
      chunk۰span l dq n ⊢
        ∃ v,
        (l +ₗ i) ↦{dq} v.

    Lemma chunk۰span𑁒update' {l} {i : Z} {dq n} j :
      (0 ≤ i ≤ j ∧ j < i + n)%Z →
      chunk۰span (l +ₗ i) dq n ⊢
        ∃ v,
        (l +ₗ j) ↦{dq} v ∗
        ( ∀ w,
          (l +ₗ j) ↦{dq} w -∗
          chunk۰span (l +ₗ i) dq n
        ).
    Lemma chunk۰span𑁒lookup𑁒acc' {l} {i : Z} {dq n} j :
      (0 ≤ i ≤ j ∧ j < i + n)%Z →
      chunk۰span (l +ₗ i) dq n ⊢
        ∃ v,
        (l +ₗ j) ↦{dq} v ∗
        ( (l +ₗ j) ↦{dq} v -∗
          chunk۰span (l +ₗ i) dq n
        ).
    Lemma chunk۰span𑁒lookup' {l} {i : Z} {dq n} j :
      (0 ≤ i ≤ j ∧ j < i + n)%Z →
      chunk۰span (l +ₗ i) dq n ⊢
        ∃ v,
        (l +ₗ j) ↦{dq} v.

    Lemma chunk۰span𑁒valid l dq n :
      0 < n →
      chunk۰span l dq n ⊢
      ⌜✓ dq⌝.
    Lemma chunk۰span𑁒combine l dq1 n1 dq2 n2 :
      n1 = n2 →
      chunk۰span l dq1 n1 -∗
      chunk۰span l dq2 n2 -∗
      chunk۰span l (dq1 ⋅ dq2) n1.
    Lemma chunk۰span𑁒valid𑁒2 l dq1 n1 dq2 n2 :
      n1 = n2 →
      0 < n1 →
      chunk۰span l dq1 n1 -∗
      chunk۰span l dq2 n2 -∗
      ⌜✓ (dq1 ⋅ dq2)⌝.
    Lemma chunk۰span𑁒dfrac𑁒ne l1 dq1 n1 l2 dq2 n2 :
      n1 = n2 →
      0 < n1 →
      ¬ ✓ (dq1 ⋅ dq2) →
      chunk۰span l1 dq1 n1 -∗
      chunk۰span l2 dq2 n2 -∗
      ⌜l1 ≠ l2⌝.
    Lemma chunk۰span𑁒ne l1 n1 l2 dq2 n2 :
      n1 = n2 →
      0 < n1 →
      chunk۰span l1 (DfracOwn 1) n1 -∗
      chunk۰span l2 dq2 n2 -∗
      ⌜l1 ≠ l2⌝.
    Lemma chunk۰span𑁒exclusive l n1 dq2 n2 :
      n1 = n2 →
      0 < n1 →
      chunk۰span l (DfracOwn 1) n1 -∗
      chunk۰span l dq2 n2 -∗
      False.
    Lemma chunk۰span𑁒persist l dq n :
      chunk۰span l dq n ⊢ |==>
      chunk۰span l DfracDiscarded n.
  End chunk۰span.

  Section chunk۰cslice.
    Implicit Type sz : nat.

    Definition chunk۰cslice l sz i dq vs : iProp Σ :=
      [∗ list] k ↦ v ∈ vs, (l +ₗ (i + k) `mod` sz) ↦{dq} v.

    #[global] Instance chunk۰cslice𑁒timeless l sz i dq vs :
      Timeless (chunk۰cslice l sz i dq vs).

    #[global] Instance chunk۰cslice𑁒persistent l sz i vs :
      Persistent (chunk۰cslice l sz i DfracDiscarded vs).

    #[global] Instance chunk۰cslice𑁒fractional l sz i vs :
      Fractional (λ q, chunk۰cslice l sz i (DfracOwn q) vs).
    #[global] Instance chunk۰cslice𑁒as_fractionak l sz i q vs :
      AsFractional (chunk۰cslice l sz i (DfracOwn q) vs) (λ q, chunk۰cslice l sz i (DfracOwn q) vs) q.

    Lemma chunk۰model𑁒to𑁒cslice l dq vs :
      chunk۰model l dq vs ⊢
      chunk۰cslice l (length vs) 0 dq vs.
    Lemma chunk𑁒model𑁒cslice𑁒cell l i sz dq v :
      chunk۰model (l +ₗ i `mod` sz) dq [v] ⊣⊢
      chunk۰cslice l sz i dq [v].

    Lemma chunk۰cslice𑁒nil l sz i dq :
      ⊢ chunk۰cslice l sz i dq [].

    Lemma chunk۰cslice𑁒singleton l sz i dq v :
      (l +ₗ i `mod` sz) ↦{dq} v ⊣⊢
      chunk۰cslice l sz i dq [v].
    Lemma chunk۰cslice𑁒singleton₁ l sz i dq v :
      (l +ₗ i `mod` sz) ↦{dq} v ⊢
      chunk۰cslice l sz i dq [v].
    Lemma chunk۰cslice𑁒singleton₂ l sz i dq v :
      chunk۰cslice l sz i dq [v] ⊢
      (l +ₗ i `mod` sz) ↦{dq} v.

    Lemma chunk۰cslice𑁒app l sz i dq vs1 vs2 :
      chunk۰cslice l sz i dq vs1 ∗
      chunk۰cslice l sz (i + length vs1) dq vs2 ⊣⊢
      chunk۰cslice l sz i dq (vs1 ++ vs2).
    Lemma chunk۰cslice𑁒app₁ l sz dq i1 vs1 i2 vs2 :
      i2 = i1 + length vs1 →
      chunk۰cslice l sz i1 dq vs1 -∗
      chunk۰cslice l sz i2 dq vs2 -∗
      chunk۰cslice l sz i1 dq (vs1 ++ vs2).
    Lemma chunk۰cslice𑁒app₂ {l sz i dq vs} vs1 vs2 :
      vs = vs1 ++ vs2 →
      chunk۰cslice l sz i dq vs ⊢
        chunk۰cslice l sz i dq vs1 ∗
        chunk۰cslice l sz (i + length vs1) dq vs2.

    Lemma chunk۰cslice𑁒app𑁒3 {l sz i dq vs} n1 i1 n2 i2 :
      i1 = i + n1 →
      i2 = i1 + n2 →
      n1 ≤ length vs →
      n1 + n2 ≤ length vs →
      chunk۰cslice l sz i dq vs ⊣⊢
        chunk۰cslice l sz i dq (take n1 vs) ∗
        chunk۰cslice l sz i1 dq (take n2 $ drop n1 vs) ∗
        chunk۰cslice l sz i2 dq (drop (n1 + n2) vs).

    Lemma chunk۰cslice𑁒cons l sz i dq v vs :
      (l +ₗ i `mod` sz) ↦{dq} v ∗
      chunk۰cslice l sz ˖i dq vs ⊣⊢
      chunk۰cslice l sz i dq (v :: vs).
    Lemma chunk۰cslice𑁒cons₁ l sz i dq v vs :
      (l +ₗ i `mod` sz) ↦{dq} v -∗
      chunk۰cslice l sz ˖i dq vs -∗
      chunk۰cslice l sz i dq (v :: vs).
    Lemma chunk۰cslice𑁒cons₂ l sz i dq v vs :
      chunk۰cslice l sz i dq (v :: vs) ⊢
        (l +ₗ i `mod` sz) ↦{dq} v ∗
        chunk۰cslice l sz ˖i dq vs.

    Lemma chunk۰cslice𑁒update {l sz i dq vs} k v :
      vs !! k = Some v →
      chunk۰cslice l sz i dq vs ⊢
        (l +ₗ ⁺(i + k) `mod` sz) ↦{dq} v ∗
        ( ∀ w,
          (l +ₗ ⁺(i + k) `mod` sz) ↦{dq} w -∗
          chunk۰cslice l sz i dq (<[k := w]> vs)
        ).
    Lemma chunk۰cslice𑁒lookup𑁒acc {l sz i dq vs} k v :
      vs !! k = Some v →
      chunk۰cslice l sz i dq vs ⊢
        (l +ₗ ⁺(i + k) `mod` sz) ↦{dq} v ∗
        ( (l +ₗ ⁺(i + k) `mod` sz) ↦{dq} v -∗
          chunk۰cslice l sz i dq vs
        ).
    Lemma chunk۰cslice𑁒lookup {l sz i dq vs} k v :
      vs !! k = Some v →
      chunk۰cslice l sz i dq vs ⊢
      (l +ₗ ⁺(i + k) `mod` sz) ↦{dq} v.

    Lemma chunk۰cslice𑁒update' {l sz i dq vs} j k v :
      (i ≤ j)%Z →
      vs !! k = Some v →
      k = ₊j - i →
      chunk۰cslice l sz i dq vs ⊢
        (l +ₗ j `mod` sz) ↦{dq} v ∗
        ( ∀ w,
          (l +ₗ j `mod` sz) ↦{dq} w -∗
          chunk۰cslice l sz i dq (<[k := w]> vs)
